pm.test("Status code is 200 OK", function () {
    pm.response.to.have.status(200);
});

pm.test("Response has expected structure and values", function () {
    const responseBody = pm.response.json();
    
    pm.expect(responseBody).to.have.property("hasResult", true);
    pm.expect(responseBody).to.have.property("result").to.be.an('object');
    
    // Add more assertions for other properties as needed
    // pm.expect(result).to.have.property("fullName").to.be.null;
    
    pm.expect(responseBody).to.have.property("resultType", 1);
    pm.expect(responseBody).to.have.property("message", "Success");
    pm.expect(responseBody).to.have.property("successful", true);
});
