pm.test("Verify response hasResult is false", function () {
    pm.expect(pm.response.json().hasResult).to.eql(false);
});

pm.test("Verify response result is null", function () {
    pm.expect(pm.response.json().result).to.be.null;
});

pm.test("Verify response resultType is 'ValidationError'", function () {
    pm.expect(pm.response.json().resultType).to.eql('ValidationError');
});

pm.test("Verify response message is 'Response has validation errors'", function () {
    pm.expect(pm.response.json().message).to.eql('Response has validation errors');
});

pm.test("Verify response validationMessages contains expected error", function () {
    const validationMessages = pm.response.json().validationMessages;

    // Check if 'validationMessages' property is an array
    pm.expect(validationMessages).to.be.an('array');
    // Check if the array contains the expected error message
    pm.expect(validationMessages).to.eql(['The rcbnNumber field is required.']);
});

pm.test("Verify response successful is false", function () {
    pm.expect(pm.response.json().successful).to.eql(false);
});
