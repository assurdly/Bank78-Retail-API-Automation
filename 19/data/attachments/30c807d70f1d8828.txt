
pm.test("Response status code is 200", function () {
  pm.response.to.have.status(200);
});


pm.test("Response has the required fields", function () {
    const responseData = pm.response.json();
    
    pm.expect(responseData).to.be.an('object');
    pm.expect(responseData.hasResult).to.exist;
    pm.expect(responseData.result).to.exist;
    pm.expect(responseData.resultType).to.exist;
    pm.expect(responseData.message).to.exist;
    pm.expect(responseData.validationMessages).to.be.null;
    pm.expect(responseData.successful).to.exist;
});


pm.test("Validate the value of hasResult is a boolean", function () {
  const responseData = pm.response.json();

  pm.expect(responseData.hasResult).to.be.a('boolean');
});


pm.test("Validate the value of resultType is an integer", function () {
    const responseData = pm.response.json();
    
    pm.expect(responseData.resultType).to.be.a('number');
});


pm.test("Validate the value of successful is a boolean", function () {
  const responseData = pm.response.json();
  
  pm.expect(responseData.successful).to.be.a('boolean');
});
