pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response has 'hasResult' property", function () {
    pm.expect(pm.response.json()).to.have.property('hasResult');
});

pm.test("Response has 'result' property", function () {
    pm.expect(pm.response.json()).to.have.property('result');
});

pm.test("Each element in the 'sections' array has 'id' property", function () {
    pm.response.json().result.sections.forEach(function(section) {
        pm.expect(section).to.have.property('id');
    });
});
