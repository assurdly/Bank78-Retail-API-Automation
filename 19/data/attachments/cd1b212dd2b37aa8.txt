pm.test("Status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Status code response is Bad Request", function () {
    pm.response.to.have.status('Bad Request');
});

var jsonData = pm.response.json();

pm.test("Response body", function () {
    pm.expect(jsonData).to.be.an('object');
    pm.expect(jsonData.hasResult).to.be.a('boolean');
    pm.expect(jsonData.result).to.be.null;
    pm.expect(jsonData.resultType).to.be.a('string');
    pm.expect(jsonData.validationMessages).to.be.an('array');
    pm.expect(jsonData.successful).to.be.a('boolean');
});

pm.test("Response has a 'hasResult' property with value 'false'", function() {
    pm.expect(jsonData).to.have.property("successful", false);
});

pm.test("Response has a 'resultType' property with value 'Success'", function() {
    pm.expect(jsonData).to.have.property("resultType", "ValidationError");
});

pm.test("Response has a 'message' property with value 'Response has validation errors'", function() {
    pm.expect(jsonData).to.have.property("message", "Response has validation errors");
});

pm.test("Response has 'validationMessages'", function() {
    pm.expect(jsonData.validationMessages[0]).to.eq('The rcbnNumber field is required.');
});

pm.test("Response has a 'successful' property with value 'false'", function() {
    pm.expect(jsonData).to.have.property("successful", false);
});
